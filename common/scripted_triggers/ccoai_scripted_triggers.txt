should_NAM_invade_europe_or_africa = {
	NAM = {
		OR = {
			is_in_faction_with = 2GB
			has_military_access_to = 2GB
		}
	}
	surrender_progress < 0.05
	#FRA = {
	#	has_capitulated = yes
	#}
	OR = {
		EUR = {
			exists = yes
			has_war_with = NAM
		}
		AFR = {
			exists = yes
			has_war_with = NAM
		}
		ASA = {
			exists = yes
			has_war_with = NAM
		}
		NAM = {
			OR = {
				AND = {
					#has_government = ROOT
					#2GB = { has_government = ROOT } 
					NOT = {
						OR = {
							EUR = { has_government = ROOT }
							AFR = { has_government = ROOT }
						}
					}
				}
				is_in_array = { potential_and_current_enemies = EUR }
				is_in_array = { potential_and_current_enemies = AFR }
				is_in_array = { potential_and_current_enemies = ASA }
			}
		}
	}
