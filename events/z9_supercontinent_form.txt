add_namespace = supercontinent_form

country_event = { # The Americas
	id = supercontinent_form.1
	is_triggered_only = yes
	hidden = yes
	option = {
		every_state = {
			limit = {
				OR = { is_on_continent = north_america is_on_continent = south_america }
				NOT = { is_core_of = ROOT }
			}
			add_core_of = ROOT
		}
	}
}
